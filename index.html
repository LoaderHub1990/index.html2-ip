// ============================================================
//  ระบบร้านค้า Discord รับซอง TrueMoney อัตโนมัติ
//  ลูกค้ากดปุ่ม → เด้ง Modal ใส่ลิงก์ซอง → รับเงินอัตโนมัติ
//  ติดตั้ง: npm install discord.js
// ============================================================
const { Client, GatewayIntentBits, ActionRowBuilder, ButtonBuilder,
  ButtonStyle, ModalBuilder, TextInputBuilder, TextInputStyle,
  EmbedBuilder, REST, Routes, SlashCommandBuilder } = require("discord.js");

const BOT_TOKEN = "ใส่_TOKEN_บอทที่นี่";
const CLIENT_ID = "ใส่_CLIENT_ID_ที่นี่";
const TM_KEY    = "ใส่_TM_KEY_จาก_ยืนยันการใช้งานรับซอง_ที่นี่";
const PHONE     = "ใส่_เบอร์_TrueMoney_ของร้าน_ที่นี่";
const API_BASE  = "http://51.68.234.157:20026";

const rest = new REST({ version:"10" }).setToken(BOT_TOKEN);
rest.put(Routes.applicationCommands(CLIENT_ID), { body:[
  new SlashCommandBuilder()
    .setName("ร้านค้า")
    .setDescription("เปิดหน้าร้านพร้อมปุ่มชำระเงินซอง TrueMoney")
    .toJSON()
]}).then(()=>console.log("✅ ลงทะเบียนคำสั่งสำเร็จ"));

const client = new Client({ intents:[GatewayIntentBits.Guilds] });
client.once("ready", ()=>console.log(`✅ Bot: ${client.user.tag}`));

client.on("interactionCreate", async interaction => {

  // /ร้านค้า → แสดงปุ่มให้ลูกค้ากด
  if (interaction.isChatInputCommand() && interaction.commandName === "ร้านค้า") {
    const row = new ActionRowBuilder().addComponents(
      new ButtonBuilder()
        .setCustomId("pay_truemoney")
        .setLabel("💰 ชำระด้วยซอง TrueMoney")
        .setStyle(ButtonStyle.Success)
    );
    await interaction.reply({
      embeds: [new EmbedBuilder()
        .setColor(0xf1c40f)
        .setTitle("🏪 ร้านค้า")
        .setDescription("กดปุ่มด้านล่างเพื่อชำระเงินด้วยซอง TrueMoney\nระบบรับซองอัตโนมัติทันที ✅")
      ],
      components: [row]
    });
  }

  // กดปุ่ม → เด้ง Modal
  if (interaction.isButton() && interaction.customId === "pay_truemoney") {
    const modal = new ModalBuilder()
      .setCustomId("modal_truemoney")
      .setTitle("💰 วางซอง TrueMoney");
    modal.addComponents(
      new ActionRowBuilder().addComponents(
        new TextInputBuilder()
          .setCustomId("voucher_url")
          .setLabel("ลิงก์ซอง TrueMoney (วางลิงก์เต็มได้เลย)")
          .setStyle(TextInputStyle.Short)
          .setPlaceholder("https://gift.truemoney.com/campaign/?v=XXXXXXXX")
          .setRequired(true)
      )
    );
    await interaction.showModal(modal);
  }

  // Modal Submit → เรียก API รับซองอัตโนมัติ
  if (interaction.isModalSubmit() && interaction.customId === "modal_truemoney") {
    await interaction.deferReply({ ephemeral: true });
    const voucherUrl = interaction.fields.getTextInputValue("voucher_url").trim();
    try {
      const apiRes = await fetch(
        `${API_BASE}/redeem?url=${encodeURIComponent(voucherUrl)}&phone=${PHONE}&apikey=${TM_KEY}`
      );
      const data = await apiRes.json();
      if (data.data?.voucher?.redeemed_amount_baht) {
        const amount = data.data.voucher.redeemed_amount_baht;
        await interaction.editReply({ embeds:[new EmbedBuilder()
          .setColor(0x2ecc71)
          .setTitle("✅ รับซองสำเร็จ!")
          .addFields(
            { name:"💵 จำนวน", value:`**${amount} บาท**`, inline:true },
            { name:"👤 ผู้ชำระ", value:`${interaction.user.tag}`, inline:true }
          ).setTimestamp()
        ]});
      } else {
        await interaction.editReply(`❌ ไม่สำเร็จ: ${data.message || data.status || "ลองใหม่อีกครั้ง"}`);
      }
    } catch(e) {
      await interaction.editReply("❌ เกิดข้อผิดพลาด กรุณาลองใหม่");
    }
  }
});

client.login(BOT_TOKEN);
